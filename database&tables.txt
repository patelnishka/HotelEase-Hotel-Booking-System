-- Hotelease Final Database Schema (2026)
SET SQL_MODE = "NO_AUTO_VALUE_ON_ZERO";
START TRANSACTION;
SET time_zone = "+00:00";

-- 1. Admin Table
CREATE TABLE `admin` (
  `a_id` INT(5) PRIMARY KEY AUTO_INCREMENT,
  `a_name` VARCHAR(20) NOT NULL,
  `a_email` VARCHAR(30) NOT NULL,
  `a_pwd` VARCHAR(255) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. Settings Table (For Invoice/Report Headers)
CREATE TABLE `settings` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `hotel_name` VARCHAR(100) NOT NULL,
  `hotel_address` TEXT NOT NULL,
  `hotel_phone` VARCHAR(20) NOT NULL,
  `hotel_email` VARCHAR(50) NOT NULL,
  `hotel_logo` VARCHAR(100) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 3. Users Table
CREATE TABLE `users` (
  `u_id` INT(5) PRIMARY KEY AUTO_INCREMENT,
  `u_name` VARCHAR(50) NOT NULL,
  `u_address` VARCHAR(255) NOT NULL,
  `u_pincode` INT(10) DEFAULT NULL,
  `u_gender` VARCHAR(10) NOT NULL,
  `u_email` VARCHAR(50) NOT NULL UNIQUE,
  `u_phone` VARCHAR(15) NOT NULL,
  `u_pwd` VARCHAR(255) NOT NULL,
  `profile_pic` VARCHAR(100) DEFAULT 'default_user.jpg',
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4. Rooms Table
CREATE TABLE `rooms` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(150) NOT NULL,
  `area` INT NOT NULL,
  `price` DECIMAL(10,2) NOT NULL,
  `quantity` INT NOT NULL,
  `adult` INT NOT NULL,
  `children` INT NOT NULL,
  `description` TEXT NOT NULL,
  `status` TINYINT(1) DEFAULT 1
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 5. Features & Room Relations
CREATE TABLE `features` (
  `f_id` int(5) PRIMARY KEY AUTO_INCREMENT,
  `f_name` varchar(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `room_features` (
  `rf_id` int(11) PRIMARY KEY AUTO_INCREMENT,
  `room_id` int(11) NOT NULL,
  `f_id` int(5) NOT NULL,
  FOREIGN KEY (`room_id`) REFERENCES `rooms` (`id`) ON DELETE CASCADE,
  FOREIGN KEY (`f_id`) REFERENCES `features` (`f_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 6. Room Numbers (Individual Units)
CREATE TABLE `room_numbers` (
  `id` INT(5) PRIMARY KEY AUTO_INCREMENT,
  `room_id` INT NOT NULL,
  `room_no` VARCHAR(50) UNIQUE NOT NULL,
  `status` ENUM('available', 'occupied', 'maintenance') DEFAULT 'available',
  FOREIGN KEY (`room_id`) REFERENCES `rooms`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 7. Images
CREATE TABLE `room_images` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `room_id` INT NOT NULL,
  `image` VARCHAR(255) NOT NULL,
  `thumb` TINYINT(1) DEFAULT 0, 
  FOREIGN KEY (`room_id`) REFERENCES `rooms`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 8. Bookings (The Heart of the System)
CREATE TABLE `bookings` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `u_id` INT NOT NULL,
  `room_id` INT NOT NULL,
  `check_in` DATE NOT NULL,
  `check_out` DATE NOT NULL,
  `total_pay` DECIMAL(10,2) NOT NULL,
  `advance` DECIMAL(10,2) NOT NULL,
  `order_id` VARCHAR(150) NOT NULL,
  `trans_id` VARCHAR(150) DEFAULT NULL,
  `status` ENUM('pending', 'booked', 'cancelled') DEFAULT 'pending',
  `arrival` TINYINT(1) NOT NULL DEFAULT 0, -- 0: Not Arrived, 1: Checked-In, 2: Checked-Out
  `room_no` VARCHAR(50) DEFAULT NULL,
  `refund` TINYINT(1) DEFAULT NULL,
  `datentime` DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`u_id`) REFERENCES `users`(`u_id`) ON DELETE CASCADE,
  FOREIGN KEY (`room_id`) REFERENCES `rooms`(`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 9. Payments (Transaction Logs)
CREATE TABLE `payments` (
  `p_id` INT PRIMARY KEY AUTO_INCREMENT,
  `booking_id` INT NOT NULL,
  `trans_id` VARCHAR(200) NOT NULL,
  `amount` DECIMAL(10,2) NOT NULL,
  `method` VARCHAR(50) DEFAULT 'Online',
  `status` VARCHAR(20) NOT NULL DEFAULT 'success',
  `payment_type` ENUM('advance', 'full', 'remaining') DEFAULT 'advance',
  `pay_date` DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`booking_id`) REFERENCES `bookings`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 10. Extras
CREATE TABLE `feedback` (
  `f_id` INT(5) PRIMARY KEY AUTO_INCREMENT,
  `u_id` INT(5) NOT NULL,
  `rating` TINYINT(1) NOT NULL,
  `message` TEXT NOT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`u_id`) REFERENCES `users`(`u_id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `contact_queries` (
  `id` INT(5) PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(30) NOT NULL,
  `email` VARCHAR(30) NOT NULL,
  `subject` VARCHAR(200) NOT NULL,
  `message` TEXT NOT NULL,
  `seen` TINYINT(1) NOT NULL DEFAULT 0,
  `date` DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
